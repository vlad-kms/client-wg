Использование:"
wg-mgr.sh [command] [options]"
command (одна из [ ${ARR_CMD} ], по-умолчанию install):"
    install    - установка пакета wireguard и других, требующихся для работы (iptables, qrencode и др.)"
        options:"
            -c, --config <filename>        - файл с данными для инсталяции"
            -r, --rules-iptables <filename>- файл с правилами iptables (ip6tables)"
            -p, --params <filename>        - создается файл с уточненными данными после инсталяции"
            -d, --hand-params <filename>   - файл созданный вручную для определения дополнительных переменных"
            -i, --wg-nic <nic_name>        - имя интерфейса для установки сервера WIREGUARD"
                --ip4 <address/mask>       - IPv4 (ip/mask) сервера для установки сервера WIREGUARD"
                --ip6 <address/mask>       - IPv6 (ip/mask) сервера для установки сервера WIREGUARD"

    uninstall  - удаление пакета wireguard и других, установленных вместе с ним, а также удалить все созданные каталоги и файлы"

    prepare    - подготовить файл с данными для инсталяции"
        options:"
            -c, --config <filename> - файл для подготовки данных для инсталяции"
            -i, --wg-nic <nic_name>        - имя интерфейса для подготовки файла настроек для дальнейшей работы скрипта"
                --ip4 <address/mask>       - IPv4 (ip/mask) сервера для подготовки файла настроек для дальнейшей работы скрипта"
                --ip6 <address/mask>       - IPv6 (ip/mask) сервера для подготовки файла настроек для дальнейшей работы скрипта"

    client     - работа с клиентами: добавить, удалить, получить список"
        options:"
            -a, --action <action>          - указывает что делать: создать клиента, удалить клиента или получить список клиентов"
                                             значение должно быть из ACTION_CLIENT (add, del, list)"
                          a | add | new   :  создать файлы настроек для клиента на сервере и клиенте, файл QRcode для клиента"
                          d | del | delete:  удалить клиента из файла настроек сервера"
                          l | list        :  получить список клиентов из файла настроек сервера"
            -p, --params <filename>        - созданный при install файл с уточненными данными используется для создания файлов клиента"
            -d, --hand-params <filename>   - файл созданный вручную для определения дополнительных переменных для настройки клиентов"
            -i, --wg-nic <nic_name>        - имя интерфейса сервера для подключения клиента"
                --ip4 <address/mask>       - IPv4 (ip/mask) клиента"
                --ip6 <address/mask>       - IPv6 (ip/mask) клиента"
            -e, --allowed-ips <network>    - список ip адресов, которым разрешен доступ в формате 1.1.1.0/24,fdoo::0/64,2.2.2.2/32"
                                             по-умолчанию только адрес клиента"
                --dns <dns1,dns2>          - список ip адресов DNS"
            -n, --name <name client>       - имя клиента"

common options:"
    -h, --help                 - описание использования скрипта"
        --debug                - вывод отладочных сообщений"
    -o, --out-path <path out>  - путь куда записываются файлы клиента,"
                                 если команда install, то создать каталог, если его нет"
                                 если команда new, то создать каталог, если его нет и записать в него файлы для клиента"
    -6, --use-ipv6             - использовать IPv6 или нет для настройки локальных адресов WIREGUARD VPN. НЕ РЕАЛИЗОВАНО (пока)"
        --dry-run              - команду не выполнять, только показать"
    -w, --wg-path <path>       - путь к установленному Wireguard"
    -f, --file-args <path>     - путь к файлу где хранятся аргументы для командной строки"
    -u, --update-args          - флаг, что надо обновить файл с аргументами соответственно текущим аргументам командной строки"
    -x, --allow-lxc            - флаг, что не блокировать установку WIREGUARD в контейнеры и VM LXD"
